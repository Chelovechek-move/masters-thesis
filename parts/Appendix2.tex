\section*{Приложение~\thesection. Аналитические выражения для матрицы $B_{\text{ft}}$}
\addcontentsline{toc}{section}{Приложение~\thesection}

В настоящем приложении приведены аналитические выражения для матрицы $B_{\text{ft}}(L_1, L_2)$. При её построении все орбитальные элементы, кроме истинной долготы 
$L$, считаются фиксированными в центральной точке манёвра $L_c = (L_1 + L_2) / 2$.
Матрица представляется в виде суммы двух слагаемых:
\begin{equation*}
    B_{\text{ft}} = B_{\text{ft}}^{(1)} + B_{\text{ft}}^{(2)},
\end{equation*}
где 
\begin{equation*}
B_{\text{ft}}^{(1)} =
    \begin{bmatrix}
        B_{\text{ft, 1}}^{(1)} &
        B_{\text{ft, 2}}^{(1)} &
        B_{\text{ft, 3}}^{(1)}
    \end{bmatrix},
\quad
B_{\text{ft}}^{(2)} =
    \begin{bmatrix}
        B_{\text{ft, 1}}^{(2)} &
        B_{\text{ft, 2}}^{(2)} &
        B_{\text{ft, 3}}^{(2)}
    \end{bmatrix}.
\end{equation*}
Представим матрицу $B_{\text{ft}}^{(1)}$ по столбцам:
\begin{equation*}
B_{\text{ft, 1}}^{(1)} =
    \begin{bmatrix}
        \frac{2a^{2}}{\sqrt{\mu P}} \left[ \frac{1}{w(L)} \right]_{L_1}^{L_2} \\
        -\sqrt{\frac{P}{\mu}} \left\{ \left[ \frac{\sin L}{w(L)} \right]_{L_1}^{L_2} -
        \frac{2k}{\eta^{3}} \left[ \Phi(L) \right]_{L_1}^{L_2} +
        \frac{k}{(1 - k) \eta^2} \left[ \Psi(L) \right]_{L_1}^{L_2} \right\} \\
        \sqrt{\frac{P}{\mu}} \left\{ -\left[ \frac{\cos L}{w(L)} \right]_{L_1}^{L_2} -
        \frac{2h}{\eta^{3}} \left[ \Phi(L) \right]_{L_1}^{L_2} +
        \frac{h}{(1 - k) \eta^2} \left[ \Psi(L) \right]_{L_1}^{L_2} \right\} \\
        0 \\
        0 \\
        \varkappa
    \end{bmatrix},
\end{equation*}
\begin{align*}
    \varkappa = &-\sqrt{\frac{P}{\mu}} b \left\{ - \frac{2e^2}{\eta^3} \left[ \Phi(L) \right]_{L_1}^{L_2} +
    \frac{1}{(1 - k) \eta^2} \left[ \Psi(L) \right]_{L_1}^{L_2} \right\} - \\
    &-2 \sqrt{\frac{P}{\mu}} \sqrt{\frac{P}{a}} \left\{
    \frac{1}{2 \eta^2} \left[ \frac{\tau(L)}{w(L)^2} \right]_{L_1}^{L_2} +
    \frac{2 + e^2}{\eta^5} \left[ \Phi(L) \right]_{L_1}^{L_2} -
    \frac{3}{2 (1 - k) \eta^4} \left[ \Psi(L) \right]_{L_1}^{L_2}
    \right\},
\end{align*}
\begin{equation*}
B_{\text{ft, 2}}^{(1)} =
    \begin{bmatrix}
        \frac{2a^{2}}{\sqrt{\mu P}} \frac{2}{\eta} \left[ \Phi(L) \right]_{L_1}^{L_2} \\
        \sqrt{\frac{P}{\mu}} \left\{ -\frac{3}{2} \left[ \frac{\cos L}{w(L)} \right]_{L_1}^{L_2} -
        \frac{1}{2} \left[ \frac{\cos L}{w(L)^2} \right]_{L_1}^{L_2} -
        \frac{3h}{\eta^3} \left[ \Phi(L) \right]_{L_1}^{L_2} +
        \frac{3h}{2 (1 - k) \eta^2} \left[ \Psi(L) \right]_{L_1}^{L_2} \right\} \\
        \sqrt{\frac{P}{\mu}} \left\{ \frac{3}{2} \left[ \frac{\sin L}{w(L)} \right]_{L_1}^{L_2} +
        \frac{1}{2} \left[ \frac{\sin L}{w(L)^2} \right]_{L_1}^{L_2} -
        \frac{3k}{\eta^3} \left[ \Phi(L) \right]_{L_1}^{L_2} +
        \frac{3k}{2 (1 - k) \eta^2} \left[ \Psi(L) \right]_{L_1}^{L_2} \right\} \\
        0 \\
        0 \\
        \sqrt{\frac{P}{\mu}} b \left\{ \left[ \frac{1}{w(L)} \right]_{L_1}^{L_2} +
        \frac{1}{2} \left[ \frac{1}{w(L)^2} \right]_{L_1}^{L_2} \right\}
    \end{bmatrix},
\end{equation*}
\begin{equation*}
B_{\text{ft, 3}}^{(1)} = \sqrt{\frac{P}{\mu}} 
    \begin{bmatrix}
        0 \\
        kI \xi \\
        -hI \xi \\
        \frac{1 + p^2 + q^2}{2} \theta \\
        \frac{1 + p^2 + q^2}{2} I \zeta \\
        I \xi
    \end{bmatrix},
\end{equation*}
\begin{align*}
    \xi = &-\frac{1}{2} \left[ \frac{\sigma(L)}{w(L)^2} \right]_{L_1}^{L_2} -
    \frac{q}{2} \left[ \frac{\cos L}{w(L)} \right]_{L_1}^{L_2} -
    \frac{Ip}{2} \left[ \frac{\sin L}{w(L)} \right]_{L_1}^{L_2} + \\
    &+\frac{\rho}{2 \eta^2} \left[ \frac{\tau(L)}{w(L)^2} \right]_{L_1}^{L_2} +
    \frac{3 \rho}{\eta^5} \left[ \Phi(L) \right]_{L_1}^{L_2} -
    \frac{\rho (4 - e^2)}{2 (1 - k) \eta^4} \left[ \Psi(L) \right]_{L_1}^{L_2},
\end{align*}
\begin{align*}
    \theta = -\frac{1}{2} \left[ \frac{\cos L}{w(L)} \right]_{L_1}^{L_2} -
    \frac{1}{2} \left[ \frac{\cos L}{w(L)^2} \right]_{L_1}^{L_2} -
    \frac{h}{2 \eta^2} \left[ \frac{\tau(L)}{w(L)^2} \right]_{L_1}^{L_2} -
    \frac{3h}{\eta^5} \left[ \Phi(L) \right]_{L_1}^{L_2} +
    \frac{h (4 - e^2)}{2 (1 - k) \eta^4} \left[ \Psi(L) \right]_{L_1}^{L_2},
\end{align*}
\begin{align*}
    \zeta = \frac{1}{2} \left[ \frac{\sin L}{w(L)} \right]_{L_1}^{L_2} +
    \frac{1}{2} \left[ \frac{\sin L}{w(L)^2} \right]_{L_1}^{L_2} -
    \frac{k}{2 \eta^2} \left[ \frac{\tau(L)}{w(L)^2} \right]_{L_1}^{L_2} -
    \frac{3k}{\eta^5} \left[ \Phi(L) \right]_{L_1}^{L_2} +
    \frac{k (4 - e^2)}{2 (1 - k) \eta^4} \left[ \Psi(L) \right]_{L_1}^{L_2}.
\end{align*}
Аналогично представим матрицу $B_{\text{ft}}^{(2)}$:
\begin{equation*}
B_{\text{ft, 1}}^{(2)} =
    \begin{bmatrix}
        \mathbf{0}_{5 \times 1} \\
        -\frac{3 a \eta^3}{\sqrt{\mu P}} \left\{ \frac{1}{2 \eta^2} \left[ \frac{\tau(L)}{w(L)^2} \right]_{L_1}^{L_2} +
        \frac{2 + e^2}{\eta^5} \left[ \Phi(L) \right]_{L_1}^{L_2} -
        \frac{3}{2 (1 - k) \eta^4} \left[ \Psi(L) \right]_{L_1}^{L_2} -
        \frac{1}{w(L_1) \eta^3} \left[ \lambda(L) \right]_{L_1}^{L_2} \right\}
    \end{bmatrix},
\end{equation*}
\begin{equation*}
B_{\text{ft, 2}}^{(2)} =
    \begin{bmatrix}
        \mathbf{0}_{5 \times 1} \\
        \varphi
    \end{bmatrix},
    \quad
    B_{\text{ft, 3}}^{(2)} = \mathbf{0}_{6 \times 1},
\end{equation*}
\begin{align*}
    \varphi = &-\frac{6 a \eta^2}{\sqrt{\mu P}} \biggl\{ \frac{1}{\eta^3} \left[ \Phi(L)^2 \right]_{L_1}^{L_2} -
    \frac{1}{(1 - k) \eta^2} \left[ \Phi(L) \Psi(L) \right]_{L_1}^{L_2} - \\
    &-\frac{1}{2 (1 - k) \eta} \left( (k - e^2) \left[ \frac{\cos L}{w(L)} \right]_{L_1}^{L_2} +
    h \left[ \frac{\sin L}{w(L)} \right]_{L_1}^{L_2} \right) - \\
    &-\frac{h}{(1 - k) \eta^2} \left[ \Phi(L) \right]_{L_1}^{L_2} +
    \frac{h}{2 (1 - k)^2 \eta} \left[ \Psi(L) \right]_{L_1}^{L_2} -
    \frac{\Phi(L_1)}{\eta^3} \left[ \lambda(L) \right]_{L_1}^{L_2} \biggr\},
\end{align*}
где
\begin{equation*}
    e = \sqrt{h^2 + k^2}, \quad
    \eta = \sqrt{1 - e^2}, \quad
    P = a (1 - e^2), \quad
    b = \frac{1}{1 + \eta},
\end{equation*}
\begin{equation*}
    \rho = -qh + Ipk, \quad
    \sigma(L) = q \cos L + I p \sin L, \quad
    \tau(L) = h \cos L - k \sin L,
\end{equation*}
\begin{equation*}
    \Phi(L) = \arctan \left( \frac{(1 - k) \tan \left(\frac{L}{2} \right) + h}{\eta} \right), \quad
    \Psi(L) = \frac{(k - e^2) \sin L - h (1 + \cos L)}{w(L)}.
\end{equation*}
